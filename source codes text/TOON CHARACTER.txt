#define FREEGLUT_STATIC
#include <GL/freeglut.h>
#include <cmath>

#define PI 3.14159f

void drawFace(float z, float r, float g, float b) {
    glColor3f(r, g, b);

    glBegin(GL_POLYGON);
    for (int i = 0; i < 360; i++) {
        float theta = i * PI / 180.0f;

        float x = 0.2f * cosf(theta);
        float y = 0.2f * sinf(theta);

        glVertex3f(x, y, z);
    }
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);

    glLoadIdentity();

    // Far character (Blue)
    glTranslatef(-0.5f, 0.0f, -2.0f);
    drawFace(-2.0f, 0.0f, 0.0f, 1.0f);

    glLoadIdentity();

    // Near character (Red)
    glTranslatef(0.5f, 0.0f, -1.0f);
    drawFace(-1.0f, 1.0f, 0.0f, 0.0f);

    glutSwapBuffers();
}

void init() {
    glEnable(GL_DEPTH_TEST);

    glClearColor(1.0f, 1.0f, 1.0f, 1.0f);

    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();

    gluPerspective(60.0, 1.0, 1.0, 10.0);

    glMatrixMode(GL_MODELVIEW);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);

    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);

    glutInitWindowSize(800, 800);

    glutCreateWindow("2D/3D Cartoon Character");

    init();

    glutDisplayFunc(display);

    glutMainLoop();

    return 0;
}

#define FREEGLUT_STATIC
#include <GL/freeglut.h>
#include <cmath>

#define PI 3.14159f

void initLighting() {
    glEnable(GL_LIGHTING);
    glEnable(GL_LIGHT0);

    GLfloat lightPos[] = { 2.0f, 2.0f, 2.0f, 1.0f };
    GLfloat lightColor[] = { 1.0f, 1.0f, 1.0f, 1.0f };

    glLightfv(GL_LIGHT0, GL_POSITION, lightPos);
    glLightfv(GL_LIGHT0, GL_DIFFUSE, lightColor);

    glEnable(GL_COLOR_MATERIAL);
    glColorMaterial(GL_FRONT, GL_AMBIENT_AND_DIFFUSE);
}

void drawSpongeBody() {
    // SpongeBob body (yellow cube)
    glColor3f(1.0f, 0.9f, 0.0f);
    glPushMatrix();
    glScalef(1.0f, 1.2f, 0.5f);
    glutSolidCube(1.0);
    glPopMatrix();
}

void drawEyes() {
    // Left eye
    glPushMatrix();
    glTranslatef(-0.2f, 0.3f, 0.3f);
    glColor3f(1, 1, 1);
    glutSolidSphere(0.1, 20, 20);

    glTranslatef(0.0f, 0.0f, 0.08f);
    glColor3f(0, 0, 0);
    glutSolidSphere(0.04, 20, 20);
    glPopMatrix();

    // Right eye
    glPushMatrix();
    glTranslatef(0.2f, 0.3f, 0.3f);
    glColor3f(1, 1, 1);
    glutSolidSphere(0.1, 20, 20);

    glTranslatef(0.0f, 0.0f, 0.08f);
    glColor3f(0, 0, 0);
    glutSolidSphere(0.04, 20, 20);
    glPopMatrix();
}

void drawMouth() {
    glColor3f(0, 0, 0);
    glBegin(GL_LINES);
    glVertex3f(-0.15f, 0.0f, 0.31f);
    glVertex3f(0.15f, 0.0f, 0.31f);
    glEnd();
}

void drawShirt() {
    glColor3f(1, 1, 1);
    glPushMatrix();
    glTranslatef(0.0f, -0.35f, 0.0f);
    glScalef(1.0f, 0.3f, 0.5f);
    glutSolidCube(1.0);
    glPopMatrix();
}

void drawShorts() {
    glColor3f(0.55f, 0.27f, 0.07f); // brown shorts
    glPushMatrix();
    glTranslatef(0.0f, -0.65f, 0.0f);
    glScalef(1.0f, 0.3f, 0.5f);
    glutSolidCube(1.0);
    glPopMatrix();
}

void drawBelt() {
    glColor3f(0, 0, 0);
    glBegin(GL_QUADS);
    glVertex3f(-0.5f, -0.55f, 0.26f);
    glVertex3f(0.5f, -0.55f, 0.26f);
    glVertex3f(0.5f, -0.45f, 0.26f);
    glVertex3f(-0.5f, -0.45f, 0.26f);
    glEnd();
}

void drawTie() {
    glColor3f(1, 0, 0); // red tie

    glBegin(GL_TRIANGLES);
    glVertex3f(0.0f, -0.35f, 0.31f);
    glVertex3f(-0.08f, -0.55f, 0.31f);
    glVertex3f(0.08f, -0.55f, 0.31f);
    glEnd();
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glLoadIdentity();

    glTranslatef(0.0f, 0.0f, -3.0f);
    glRotatef(20, 1, 1, 0); // nice 3D angle

    drawSpongeBody();
    drawEyes();
    drawMouth();
    drawShirt();
    drawBelt();
    drawShorts();
    drawTie();

    glutSwapBuffers();
}

void init() {
    glEnable(GL_DEPTH_TEST);

    glClearColor(0.6f, 0.8f, 1.0f, 1.0f); // sky background

    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluPerspective(60.0, 1.0, 1.0, 10.0);

    glMatrixMode(GL_MODELVIEW);

    initLighting();
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);
    glutInitWindowSize(800, 800);
    glutCreateWindow("3D SpongeBob-like Character");

    init();

    glutDisplayFunc(display);
    glutMainLoop();

    return 0;
}

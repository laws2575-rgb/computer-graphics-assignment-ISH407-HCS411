#define FREEGLUT_STATIC
#include <GL/freeglut.h>
#include <cmath>

#define PI 3.14159f

// Animation states
float legAngle = 0.0f;
float armAngle = 0.0f;
float blinkTimer = 0.0f;
bool eyesClosed = false;
float bubbleY = -1.0f;

// ---------------- BACKGROUND ----------------
void drawBackground() {
    glDisable(GL_LIGHTING);

    // underwater gradient sky
    glBegin(GL_QUADS);
    glColor3f(0.0f, 0.6f, 0.8f);
    glVertex3f(-5, -5, -5);
    glVertex3f(5, -5, -5);
    glColor3f(0.0f, 0.3f, 0.6f);
    glVertex3f(5, 5, -5);
    glVertex3f(-5, 5, -5);
    glEnd();
}

// ---------------- SPONGE BODY ----------------
void drawBody() {
    glColor3f(1.0f, 0.9f, 0.0f);
    glPushMatrix();
    glScalef(1.0f, 1.2f, 0.6f);
    glutSolidCube(1.0);
    glPopMatrix();
}

// ---------------- EYES ----------------
void drawEyes() {
    if (eyesClosed) {
        glColor3f(0, 0, 0);
        glBegin(GL_LINES);
        glVertex3f(-0.2f, 0.3f, 0.31f);
        glVertex3f(-0.05f, 0.3f, 0.31f);

        glVertex3f(0.2f, 0.3f, 0.31f);
        glVertex3f(0.05f, 0.3f, 0.31f);
        glEnd();
        return;
    }

    // left eye
    glPushMatrix();
    glTranslatef(-0.2f, 0.3f, 0.3f);
    glColor3f(1, 1, 1);
    glutSolidSphere(0.1, 20, 20);

    glTranslatef(0, 0, 0.08f);
    glColor3f(0, 0, 0);
    glutSolidSphere(0.04, 20, 20);
    glPopMatrix();

    // right eye
    glPushMatrix();
    glTranslatef(0.2f, 0.3f, 0.3f);
    glColor3f(1, 1, 1);
    glutSolidSphere(0.1, 20, 20);

    glTranslatef(0, 0, 0.08f);
    glColor3f(0, 0, 0);
    glutSolidSphere(0.04, 20, 20);
    glPopMatrix();
}

// ---------------- MOUTH ----------------
void drawMouth() {
    glColor3f(0, 0, 0);
    glBegin(GL_LINES);
    glVertex3f(-0.15f, 0.0f, 0.31f);
    glVertex3f(0.15f, 0.0f, 0.31f);
    glEnd();
}

// ---------------- CLOTHES ----------------
void drawShirt() {
    glColor3f(1, 1, 1);
    glPushMatrix();
    glTranslatef(0, -0.35f, 0);
    glScalef(1, 0.3f, 0.6f);
    glutSolidCube(1.0);
    glPopMatrix();
}

void drawShorts() {
    glColor3f(0.55f, 0.27f, 0.07f);
    glPushMatrix();
    glTranslatef(0, -0.65f, 0);
    glScalef(1, 0.3f, 0.6f);
    glutSolidCube(1.0);
    glPopMatrix();
}

void drawBelt() {
    glColor3f(0, 0, 0);
    glBegin(GL_QUADS);
    glVertex3f(-0.5f, -0.55f, 0.31f);
    glVertex3f(0.5f, -0.55f, 0.31f);
    glVertex3f(0.5f, -0.45f, 0.31f);
    glVertex3f(-0.5f, -0.45f, 0.31f);
    glEnd();
}

void drawTie() {
    glColor3f(1, 0, 0);
    glBegin(GL_TRIANGLES);
    glVertex3f(0, -0.35f, 0.32f);
    glVertex3f(-0.08f, -0.55f, 0.32f);
    glVertex3f(0.08f, -0.55f, 0.32f);
    glEnd();
}

// ---------------- LIMBS (ANIMATED) ----------------
void drawLimbs() {
    // LEFT ARM
    glPushMatrix();
    glTranslatef(-0.6f, 0.1f, 0);
    glRotatef(armAngle, 1, 0, 0);
    glColor3f(1.0f, 0.9f, 0.0f);
    glScalef(0.1f, 0.5f, 0.1f);
    glutSolidCube(1);
    glPopMatrix();

    // RIGHT ARM
    glPushMatrix();
    glTranslatef(0.6f, 0.1f, 0);
    glRotatef(-armAngle, 1, 0, 0);
    glColor3f(1.0f, 0.9f, 0.0f);
    glScalef(0.1f, 0.5f, 0.1f);
    glutSolidCube(1);
    glPopMatrix();

    // LEFT LEG
    glPushMatrix();
    glTranslatef(-0.2f, -1.0f, 0);
    glRotatef(legAngle, 1, 0, 0);
    glColor3f(0, 0, 0);
    glScalef(0.15f, 0.5f, 0.15f);
    glutSolidCube(1);
    glPopMatrix();

    // RIGHT LEG
    glPushMatrix();
    glTranslatef(0.2f, -1.0f, 0);
    glRotatef(-legAngle, 1, 0, 0);
    glColor3f(0, 0, 0);
    glScalef(0.15f, 0.5f, 0.15f);
    glutSolidCube(1);
    glPopMatrix();
}

// ---------------- BUBBLES ----------------
void drawBubble() {
    glPushMatrix();
    glTranslatef(-1.5f, bubbleY, 0);
    glColor3f(1, 1, 1);
    glutSolidSphere(0.1, 15, 15);
    glPopMatrix();
}

// ---------------- DISPLAY ----------------
void display() {
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glLoadIdentity();

    drawBackground();

    glTranslatef(0, 0, -4);
    glRotatef(15, 1, 1, 0);

    drawBody();
    drawEyes();
    drawMouth();
    drawShirt();
    drawBelt();
    drawShorts();
    drawTie();
    drawLimbs();
    drawBubble();

    glutSwapBuffers();
}

// ---------------- ANIMATION ----------------
void update(int value) {
    legAngle = sin(glutGet(GLUT_ELAPSED_TIME) * 0.005f) * 30;
    armAngle = sin(glutGet(GLUT_ELAPSED_TIME) * 0.005f) * 25;

    // blinking
    blinkTimer += 1;
    if (blinkTimer > 100) {
        eyesClosed = true;
    }
    if (blinkTimer > 120) {
        eyesClosed = false;
        blinkTimer = 0;
    }

    // bubbles rising
    bubbleY += 0.02f;
    if (bubbleY > 2.5f) bubbleY = -1.5f;

    glutPostRedisplay();
    glutTimerFunc(16, update, 0);
}

// ---------------- INIT ----------------
void init() {
    glEnable(GL_DEPTH_TEST);

    glClearColor(0.0f, 0.5f, 0.8f, 1.0f);

    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluPerspective(60, 1, 1, 10);

    glMatrixMode(GL_MODELVIEW);
}

// ---------------- MAIN ----------------
int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);
    glutInitWindowSize(800, 800);
    glutCreateWindow("Animated SpongeBob 3D");

    init();

    glutDisplayFunc(display);
    glutTimerFunc(0, update, 0);

    glutMainLoop();
    return 0;
}

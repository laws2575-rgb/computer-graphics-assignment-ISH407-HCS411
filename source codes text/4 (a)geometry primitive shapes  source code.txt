#define FREEGLUT_STATIC
#include <GL/freeglut.h>

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    // GL_POINTS
    glPointSize(6);
    glBegin(GL_POINTS);
    glColor3f(1, 0, 0);
    glVertex2f(-0.9, 0.8);
    glVertex2f(-0.85, 0.85);
    glVertex2f(-0.8, 0.8);
    glEnd();

    // GL_LINES
    glBegin(GL_LINES);
    glColor3f(0, 1, 0);
    glVertex2f(-0.7, 0.8);
    glVertex2f(-0.5, 0.9);
    glVertex2f(-0.7, 0.9);
    glVertex2f(-0.5, 0.8);
    glEnd();

    // GL_LINE_STRIP
    glBegin(GL_LINE_STRIP);
    glColor3f(0, 0, 1);
    glVertex2f(-0.4, 0.8);
    glVertex2f(-0.3, 0.9);
    glVertex2f(-0.2, 0.8);
    glVertex2f(-0.1, 0.9);
    glEnd();

    // GL_LINE_LOOP
    glBegin(GL_LINE_LOOP);
    glColor3f(1, 1, 0);
    glVertex2f(0.0, 0.8);
    glVertex2f(0.1, 0.9);
    glVertex2f(0.2, 0.8);
    glVertex2f(0.1, 0.7);
    glEnd();

    // GL_TRIANGLES
    glBegin(GL_TRIANGLES);
    glColor3f(1, 0, 1);
    glVertex2f(-0.9, 0.4);
    glVertex2f(-0.7, 0.4);
    glVertex2f(-0.8, 0.6);
    glEnd();

    // GL_TRIANGLE_STRIP
    glBegin(GL_TRIANGLE_STRIP);
    glColor3f(0, 1, 1);
    glVertex2f(-0.5, 0.4);
    glVertex2f(-0.4, 0.6);
    glVertex2f(-0.3, 0.4);
    glVertex2f(-0.2, 0.6);
    glEnd();

    // GL_TRIANGLE_FAN
    glBegin(GL_TRIANGLE_FAN);
    glColor3f(0.8, 0.3, 0.1);
    glVertex2f(0.1, 0.5);
    glVertex2f(0.0, 0.4);
    glVertex2f(0.1, 0.6);
    glVertex2f(0.2, 0.4);
    glVertex2f(0.3, 0.6);
    glEnd();

    // GL_QUADS
    glBegin(GL_QUADS);
    glColor3f(0.3, 0.8, 0.4);
    glVertex2f(-0.9, 0.0);
    glVertex2f(-0.7, 0.0);
    glVertex2f(-0.7, 0.2);
    glVertex2f(-0.9, 0.2);
    glEnd();

    // GL_QUAD_STRIP
    glBegin(GL_QUAD_STRIP);
    glColor3f(0.9, 0.5, 0.2);
    glVertex2f(-0.5, 0.0);
    glVertex2f(-0.5, 0.2);
    glVertex2f(-0.4, 0.0);
    glVertex2f(-0.4, 0.2);
    glVertex2f(-0.3, 0.0);
    glVertex2f(-0.3, 0.2);
    glEnd();

    // GL_POLYGON
    glBegin(GL_POLYGON);
    glColor3f(0.5, 0.2, 0.9);
    glVertex2f(0.1, 0.0);
    glVertex2f(0.2, 0.1);
    glVertex2f(0.3, 0.0);
    glVertex2f(0.25, -0.1);
    glVertex2f(0.15, -0.1);
    glEnd();

    glFlush();
}

void init() {
    glClearColor(0, 0, 0, 1);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(1200, 800);
    glutCreateWindow("OpenGL Geometric Primitives");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
